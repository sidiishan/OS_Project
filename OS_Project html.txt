  <script>
    // ---------- OS-Level Data Structures ----------
    let contactDatabase = [];   // each: { id, number, name, spamScore }
    let callHistory = [];       // each: { timestamp, callerNumber, resolvedName, spamScore, verdict, blocked }

    // Metrics
    let stats = {
      identified: 0,
      spamBlocked: 0,
      unknownCallers: 0,
      totalSpamScoreSum: 0,
      totalCallsProcessed: 0
    };

    // Helper: update contact stats & registry UI
    function updateContactStatsUI() {
      const total = contactDatabase.length;
      const spamEntries = contactDatabase.filter(c => c.spamScore >= 7).length;
      const safeEntries = total - spamEntries;
      document.getElementById("totalContactsStat").innerText = total;
      document.getElementById("spamEntriesStat").innerText = spamEntries;
      document.getElementById("safeEntriesStat").innerText = safeEntries;
      renderContactTable();
    }

    function renderContactTable() {
      let container = document.getElementById("contactListDynamic");
      if (!container) {
        const registryDiv = document.querySelector(".truecaller-glass.rounded-2xl.p-5");
        if (registryDiv) {
          let existing = document.getElementById("contactListDynamic");
          if (!existing) {
            const div = document.createElement("div");
            div.id = "contactListDynamic";
            div.className = "mt-4 max-h-40 overflow-y-auto text-xs";
            registryDiv.appendChild(div);
            container = div;
          } else container = existing;
        }
      }
      if (container) {
        if (contactDatabase.length === 0) {
          container.innerHTML = '<div class="text-slate-500 italic text-center py-2">📭 No contacts registered</div>';
          return;
        }
        let html = '<div class="space-y-1 mt-2"><div class="grid grid-cols-3 gap-1 text-[10px] text-slate-400 border-b border-slate-700 pb-1"><span>Number</span><span>Name</span><span>SpamScore</span></div>';
        contactDatabase.slice(0, 8).forEach(c => {
          html += `<div class="grid grid-cols-3 gap-1 py-1 border-b border-slate-800/50"><span class="truncate">${c.number}</span><span>${c.name}</span><span class="${c.spamScore>=7?'text-red-400':'text-emerald-300'}">${c.spamScore}</span></div>`;
        });
        if (contactDatabase.length > 8) html += `<div class="text-[10px] text-slate-500 text-center">+${contactDatabase.length-8} more</div>`;
        html += '</div>';
        container.innerHTML = html;
      }
    }

    function registerContact(number, name, spamScore) {
      if (!number.trim() || !name.trim()) return false;
      const exists = contactDatabase.find(c => c.number === number);
      if (exists) {
        alert("Caller ID already exists. Use different number or edit.");
        return false;
      }
      const spamVal = Math.min(10, Math.max(0, parseInt(spamScore) || 0));
      contactDatabase.push({ id: Date.now() + Math.random(), number: number.trim(), name: name.trim(), spamScore: spamVal });
      updateContactStatsUI();
      return true;
    }

    // OS-level caller identification + spam detection (Scheduler simulation)
    function processIncomingCall(callerNumber) {
      if (!callerNumber.trim()) {
        alert("Enter a valid caller number");
        return null;
      }
      const contact = contactDatabase.find(c => c.number === callerNumber);
      let resolvedName = null;
      let spamScore = 5;   // default medium risk for unknown
      let identifiedFlag = false;

      if (contact) {
        resolvedName = contact.name;
        spamScore = contact.spamScore;
        identifiedFlag = true;
      } else {
        resolvedName = "Unknown Caller";
        spamScore = 6;   // unknown default moderate-high
        identifiedFlag = false;
      }
      
      const blocked = (spamScore >= 7);
      let verdictMsg = "";
      if (blocked) {
        verdictMsg = `🚫 BLOCKED (spam score ${spamScore}/10) · Kernel preemptive kill`;
      } else if (spamScore >= 4) {
        verdictMsg = `⚠️ MEDIUM RISK · Score ${spamScore}/10 · Show cautiously`;
      } else {
        verdictMsg = `✅ SAFE · Trusted caller (score ${spamScore}) · Ring allowed`;
      }
      
      // update stats
      stats.totalCallsProcessed++;
      if (identifiedFlag) stats.identified++;
      if (blocked) stats.spamBlocked++;
      if (!identifiedFlag && !contact) stats.unknownCallers++;
      stats.totalSpamScoreSum += spamScore;
      
      const timestamp = new Date().toLocaleTimeString([], { hour:'2-digit', minute:'2-digit', second:'2-digit' });
      const logEntry = {
        timestamp,
        callerNumber,
        resolvedName: resolvedName || (contact?.name || "Unknown"),
        spamScore,
        verdict: verdictMsg,
        blocked,
        identified: identifiedFlag
      };
      callHistory.unshift(logEntry);
      updateCallLogUI();
      updateMetricsUI();
      
      document.getElementById("callCounter").innerText = stats.totalCallsProcessed;
      document.getElementById("callsProcessedStat").innerText = stats.totalCallsProcessed;
      return logEntry;
    }

    function updateCallLogUI() {
      const container = document.getElementById("callLogContainer");
      if (!container) return;
      if (callHistory.length === 0) {
        container.innerHTML = '<div class="text-center text-slate-500 text-sm py-6">📞 No incoming calls yet. Use call simulator above.</div>';
        return;
      }
      let html = '';
      callHistory.slice(0, 30).forEach(call => {
        const bgClass = call.blocked ? 'bg-red-950/30 border-l-4 border-red-500' : (call.spamScore >= 4 ? 'bg-amber-950/20 border-l-4 border-amber-500' : 'bg-emerald-950/20 border-l-4 border-emerald-500');
        const badgeClass = call.blocked ? 'spam-badge text-white text-[10px] px-2 py-0.5 rounded-full' : 'safe-badge text-white text-[10px] px-2 py-0.5 rounded-full';
        const badgeText = call.blocked ? '🚫 BLOCKED' : (call.spamScore >= 4 ? '⚠️ REVIEW' : '🟢 ALLOW');
        html += `
          <div class="call-card p-3 rounded-xl ${bgClass} bg-slate-800/40 backdrop-blur-sm flex flex-wrap justify-between items-start gap-2">
            <div class="flex-1">
              <div class="flex items-center gap-2 flex-wrap">
                <span class="font-mono text-sm font-bold">${escapeHtml(call.callerNumber)}</span>
                <span class="text-xs text-teal-300">➡️ ${escapeHtml(call.resolvedName)}</span>
                <span class="${badgeClass} text-[10px] font-semibold">${badgeText}</span>
              </div>
              <div class="text-xs text-slate-300 mt-1 flex gap-3 flex-wrap">
                <span>📊 SpamScore: ${call.spamScore}/10</span>
                <span class="text-slate-400">🕒 ${call.timestamp}</span>
              </div>
              <div class="text-[11px] text-slate-300 mt-1">${escapeHtml(call.verdict)}</div>
            </div>
            <div class="text-[10px] text-slate-400 font-mono">OS: ${call.blocked ? 'KILLED' : 'SCHEDULED'}</div>
          </div>
        `;
      });
      container.innerHTML = html;
    }
    
    function escapeHtml(str) { return str.replace(/[&<>]/g, function(m){if(m==='&') return '&amp;'; if(m==='<') return '&lt;'; if(m==='>') return '&gt;'; return m;}); }

    function updateMetricsUI() {
      document.getElementById("identifiedCount").innerText = stats.identified;
      document.getElementById("spamBlockedCount").innerText = stats.spamBlocked;
      document.getElementById("unknownCount").innerText = stats.unknownCallers;
      const avgSpam = stats.totalCallsProcessed === 0 ? 0 : (stats.totalSpamScoreSum / stats.totalCallsProcessed);
      document.getElementById("avgSpamScoreMetric").innerText = avgSpam.toFixed(2);
      document.getElementById("callsProcessedStat").innerText = stats.totalCallsProcessed;
      document.getElementById("callCounter").innerText = stats.totalCallsProcessed;
      document.getElementById("totalCallsMetric").innerText = stats.totalCallsProcessed;
    }

    function resetFullSimulation() {
      contactDatabase = [];
      callHistory = [];
      stats = { identified: 0, spamBlocked: 0, unknownCallers: 0, totalSpamScoreSum: 0, totalCallsProcessed: 0 };
      updateContactStatsUI();
      updateCallLogUI();
      updateMetricsUI();
      document.getElementById("callCounter").innerText = "0";
      document.getElementById("callsProcessedStat").innerText = "0";
      const dynamicList = document.getElementById("contactListDynamic");
      if (dynamicList) dynamicList.innerHTML = '<div class="text-slate-500 italic text-center py-2">📭 No contacts registered</div>';
    }

    function exportCallLogsCSV() {
      if (callHistory.length === 0) { alert("No call logs to export."); return; }
      const headers = ["Timestamp", "Caller Number", "Resolved Name", "Spam Score", "Verdict", "Blocked"];
      const rows = callHistory.map(c => [c.timestamp, c.callerNumber, c.resolvedName, c.spamScore, c.verdict, c.blocked ? "Yes" : "No"]);
      const csvContent = [headers, ...rows].map(row => row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(",")).join("\n");
      const blob = new Blob([csvContent], {type: "text/csv;charset=utf-8;"});
      const link = document.createElement("a");
      const url = URL.createObjectURL(blob);
      link.href = url;
      link.setAttribute("download", "truecaller_os_log.csv");
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      URL.revokeObjectURL(url);
    }

    function autoCallBurst() {
      const preset = document.getElementById("callTypePreset").value;
      let numbersToCall = [];
      if (preset === "random" && contactDatabase.length > 0) {
        const randomContact = contactDatabase[Math.floor(Math.random() * contactDatabase.length)];
        numbersToCall.push(randomContact.number);
        numbersToCall.push("+91" + Math.floor(Math.random() * 9000000000));
      } else if (preset === "trusted" && contactDatabase.length > 0) {
        const trusted = contactDatabase.filter(c => c.spamScore < 4);
        if (trusted.length) numbersToCall.push(trusted[0].number);
        else numbersToCall.push("+1800SAFE123");
      } else {
